Manipulação de Strings

strip e split

texto = "  Olá, mundo!  "
print(texto.strip())  # "Olá, mundo!"

texto = "  Python   "
print(texto.strip(" "))  # "Python"

frase = "maçã,banana,uva"
print(frase.split(","))  # ["maçã", "banana", "uva"]

frase = "um dois três quatro"
print(frase.split())  # ["um", "dois", "três", "quatro"]

startswith e replace

mensagem = "Erro: Arquivo não encontrado"
print(mensagem.startswith("Erro"))  # True

mensagem = "Aviso: Baixa bateria"
print(mensagem.startswith("Erro"))  # False

texto = "Olá, mundo!"
novo_texto = texto.replace("mundo", "Python")
print(novo_texto)  # "Olá, Python!"

frase = "banana, maçã, banana"
print(frase.replace("banana", "laranja"))  # "laranja, maçã, laranja"

Argumentos de Linha de Comando

argparse

import argparse

parser = argparse.ArgumentParser(description="Exemplo de argparse")
parser.add_argument("nome", type=str, help="Nome do usuário")
parser.add_argument("idade", type=int, help="Idade do usuário")
args = parser.parse_args()
print(f"Olá, {args.nome}! Você tem {args.idade} anos.")

Estruturas de Dados

Listas e append

numeros = [1, 2, 3]
numeros.append(4)
print(numeros)  # [1, 2, 3, 4]

nomes = []
nomes.append("Ana")
nomes.append("Carlos")
print(nomes)  # ["Ana", "Carlos"]

Loops

Loop for

for i in range(5):
    print(f"Iteração {i}")

lista = ["maçã", "banana", "uva"]
for fruta in lista:
    print(fruta)

Loop while

contador = 0
while contador < 3:
    print(f"Contagem: {contador}")
    contador += 1

Manipulação de Arquivos

Abrindo e lendo arquivos

with open("exemplo.txt", "r") as arquivo:
    for linha in arquivo:
        print(linha.strip())

Escrevendo em arquivos

with open("saida.txt", "w") as arquivo:
    arquivo.write("Este é um teste!\n")
    arquivo.write("Mais uma linha de teste.")
