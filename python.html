Manipulação de Strings

Removendo Espaços e Separando Palavras

Strings são sequências de caracteres em Python e oferecem diversos métodos úteis para manipulação.

texto = "  Olá, mundo!  "
print(texto.strip())  # "Olá, mundo!"
print(texto.lstrip())  # "Olá, mundo!  "
print(texto.rstrip())  # "  Olá, mundo!"
frase = "maçã,banana,uva"
print(frase.split(","))  # ["maçã", "banana", "uva"]
frase2 = "Python é incrível"
print(frase2.split())  # ["Python", "é", "incrível"]

Formatando Strings

nome = "Ana"
idade = 23
print(f"Meu nome é {nome} e eu tenho {idade} anos.")
print("Meu nome é {} e eu tenho {} anos.".format(nome, idade))
print("Meu nome é %s e eu tenho %d anos." % (nome, idade))

Estruturas de Dados

Listas

lista = [1, 2, 3, 4]
lista.append(5)
lista.remove(3)
print(lista[0])
for item in lista:
    print(item)

Tuplas

coordenadas = (10, 20)
print(coordenadas[0])
pessoa = ("Ana", 23, "Engenheira")
print(pessoa)

Dicionários

usuario = {"nome": "Ana", "idade": 23, "cidade": "Caen"}
print(usuario["nome"])
usuario["profissao"] = "Engenheira"
usuario.pop("idade")
for chave, valor in usuario.items():
    print(f"{chave}: {valor}")

Loops

Loop For

for i in range(5):
    print(f"Iteração {i}")
for fruta in ["maçã", "banana", "uva"]:
    print(fruta)
for chave, valor in {"nome": "Ana", "idade": 23}.items():
    print(f"{chave}: {valor}")

Loop While

contador = 0
while contador < 5:
    print(f"Contador: {contador}")
    contador += 1

Manipulação de Arquivos

Lendo Arquivos

with open("dados.txt", "r") as arquivo:
    conteudo = arquivo.read()
    print(conteudo)

Escrevendo em Arquivos

with open("saida.txt", "w") as arquivo:
    arquivo.write("Hello, World!\n")

Adicionando Conteúdo a um Arquivo

with open("saida.txt", "a") as arquivo:
    arquivo.write("Nova linha adicionada!\n")

Lendo Linha por Linha

with open("saida.txt", "r") as arquivo:
    for linha in arquivo:
        print(linha.strip())
